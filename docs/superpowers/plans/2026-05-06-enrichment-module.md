# Enrichment Module Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement the third-party URL enrichment module in mx-core ŌĆö a provider framework that resolves URLs to structured metadata, stores results in PostgreSQL, and exposes a public API for frontends.

**Architecture:** NestJS module with a provider registry pattern. Each provider (GitHub, TMDB, Bangumi, etc.) implements a common interface. URLs are matched to providers, fetched, normalized into a universal `EnrichmentResult` shape, cached in PG + Redis, and served via a public API. Config credentials are managed centrally via the existing configs system.

**Tech Stack:** NestJS, Drizzle ORM (PostgreSQL), Redis, Zod, TypeScript

```typescript
import { Injectable, Logger, Type } from '@nestjs/common'

import type { EnrichmentProvider, ProviderMeta } from '../enrichment.types'
import type { UrlMatchResult } from '../enrichment.types'
import { ENRICHMENT_PROVIDER_TOKEN } from './provider.interface'

@Injectable()
export class ProviderRegistry {
  private readonly logger = new Logger(ProviderRegistry.name)
  private providers: EnrichmentProvider[] = []

  register(provider: EnrichmentProvider) {
    this.providers.push(provider)
    // Sort by priority descending (highest priority first)
    this.providers.sort((a, b) => b.priority - a.priority)
    this.logger.log(`Registered enrichment provider: ${provider.name} (priority: ${provider.priority})`)
  }

  match(url: URL): { provider: EnrichmentProvider; match: UrlMatchResult } | null {
    for (const provider of this.providers) {
      const result = provider.matchUrl(url)
      if (result) {
        return { provider, match: result }
      }
    }
    return null
  }

  getByName(name: string): EnrichmentProvider | undefined {
    return this.providers.find((p) => p.name === name)
  }

  getAllProviders(): EnrichmentProvider[] {
    return [...this.providers]
  }

  getCategories(): string[] {
    return [...new Set(this.providers.map((p) => p.category))]
  }

  getProviderMetas(enabledChecker: (provider: EnrichmentProvider) => boolean): ProviderMeta[] {
    return this.providers.map((p) => ({
      name: p.name,
      displayName: p.displayName,
      category: p.category,
      enabled: enabledChecker(p),
      requiredConfigKeys: p.requiredConfigKeys,
      featureGateConfigKey: p.featureGateConfigKey,
    }))
  }
}
```

```typescript
import { Injectable } from '@nestjs/common'
import { eq, and, sql, lt, gt } from 'drizzle-orm'

import type { EntityId } from '~/shared/id/entity-id'
import { Inject } from '@nestjs/common'
import { PG_DB_TOKEN } from '~/constants/system.constant'
import type { AppDatabase } from '~/processors/database/postgres.provider'
import { BaseRepository } from '~/processors/database/base.repository'
import type { EnrichmentRow, EnrichmentResult } from './enrichment.types'

@Injectable()
export class EnrichmentRepository extends BaseRepository {
  constructor(
    @Inject(PG_DB_TOKEN) db: AppDatabase,
  ) {
    super(db)
  }

  async findByProviderAndExternalId(
    provider: string,
    externalId: string,
  ): Promise<EnrichmentRow | null> {
    // Import the table dynamically to avoid circular imports
    const { enrichmentCacheTable } = await import('@mx-space/db-schema/schema')
    const rows = await this.db
      .select()
      .from(enrichmentCacheTable)
      .where(
        and(
          eq(enrichmentCacheTable.provider, provider),
          eq(enrichmentCacheTable.externalId, externalId),
        ),
      )
      .limit(1)
    return rows[0] ? this.mapRow(rows[0]) : null
  }

  async findByUrl(url: string): Promise<EnrichmentRow | null> {
    const { enrichmentCacheTable } = await import('@mx-space/db-schema/schema')
    const rows = await this.db
      .select()
      .from(enrichmentCacheTable)
      .where(eq(enrichmentCacheTable.url, url))
      .limit(1)
    return rows[0] ? this.mapRow(rows[0]) : null
  }

  async upsert(
    provider: string,
    externalId: string,
    url: string,
    normalized: EnrichmentResult,
    raw: unknown | null,
    expiresAt: Date | null,
  ): Promise<EnrichmentRow> {
    const { enrichmentCacheTable } = await import('@mx-space/db-schema/schema')
    const { default: SnowflakeService } = await import('~/processors/database/snowflake.service')
    // Use raw SQL for upsert via drizzle
    const result = await this.db
      .insert(enrichmentCacheTable)
      .values({
        provider,
        externalId,
        url,
        normalized,
        raw,
        expiresAt,
        failureCount: 0,
        lastError: null,
      })
      .onConflictDoUpdate({
        target: [enrichmentCacheTable.provider, enrichmentCacheTable.externalId],
        set: {
          normalized,
          raw,
          url,
          expiresAt,
          failureCount: 0,
          lastError: null,
          fetchedAt: new Date(),
        },
      })
      .returning()

    return this.mapRow(result[0])
  }

  async upsertIfAbsent(input: {
    provider: string
    externalId: string
    url: string
    normalized: EnrichmentResult
    raw: unknown | null
    fetchedAt: Date
    expiresAt: Date | null
  }): Promise<void> {
    const { enrichmentCacheTable } = await import('@mx-space/db-schema/schema')
    const existing = await this.findByProviderAndExternalId(input.provider, input.externalId)
    if (existing) return

    await this.db.insert(enrichmentCacheTable).values({
      provider: input.provider,
      externalId: input.externalId,
      url: input.url,
      normalized: input.normalized,
      raw: input.raw,
      expiresAt: input.expiresAt,
      failureCount: 0,
      lastError: null,
    })
  }

  async recordFailure(
    provider: string,
    externalId: string,
    error: string,
  ): Promise<void> {
    const { enrichmentCacheTable } = await import('@mx-space/db-schema/schema')
    await this.db
      .update(enrichmentCacheTable)
      .set({
        failureCount: sql`${enrichmentCacheTable.failureCount} + 1`,
        lastError: error,
      })
      .where(
        and(
          eq(enrichmentCacheTable.provider, provider),
          eq(enrichmentCacheTable.externalId, externalId),
        ),
      )
  }

  async deleteByProviderAndExternalId(
    provider: string,
    externalId: string,
  ): Promise<void> {
    const { enrichmentCacheTable } = await import('@mx-space/db-schema/schema')
    await this.db
      .delete(enrichmentCacheTable)
      .where(
        and(
          eq(enrichmentCacheTable.provider, provider),
          eq(enrichmentCacheTable.externalId, externalId),
        ),
      )
  }

  async listPaginated(page: number, size: number) {
    const { enrichmentCacheTable } = await import('@mx-space/db-schema/schema')
    const offset = (page - 1) * size
    const rows = await this.db
      .select()
      .from(enrichmentCacheTable)
      .orderBy(sql`${enrichmentCacheTable.createdAt} DESC`)
      .limit(size)
      .offset(offset)

    const countResult = await this.db
      .select({ count: sql<number>`count(*)` })
      .from(enrichmentCacheTable)

    const total = Number(countResult[0].count)
    return {
      data: rows.map((r) => this.mapRow(r)),
      pagination: this.paginationOf(total, page, size),
    }
  }

  private mapRow(row: any): EnrichmentRow {
    return {
      id: row.id,
      provider: row.provider,
      externalId: row.externalId,
      url: row.url,
      normalized: row.normalized as EnrichmentResult,
      raw: row.raw,
      fetchedAt: row.fetchedAt,
      expiresAt: row.expiresAt,
      failureCount: row.failureCount,
      lastError: row.lastError,
      createdAt: row.createdAt,
    }
  }
}
```

3. Check provider enabled/token
4. DB cache hit (not expired) ŌåÆ write Redis ŌåÆ return
5. DB cache hit (expired) + backoff ŌåÆ return stale
6. Provider fetch ŌåÆ upsert DB ŌåÆ write Redis ŌåÆ return
7. Fetch failure ŌåÆ upsert failureCount ŌåÆ return stale or 502

```typescript
import { Injectable, Logger } from '@nestjs/common'
import { createHash } from 'node:crypto'

import { ConfigsService } from '~/modules/configs/configs.service'
import { RedisService } from '~/processors/redis/redis.service'

import type {
  EnrichmentResult,
  EnrichmentRow,
  ProviderMeta,
} from './enrichment.types'
import {
  ProviderDisabledError,
  TokenMissingError,
} from './enrichment.types'
import { EnrichmentRepository } from './enrichment.repository'
import { ProviderRegistry } from './providers/provider.registry'

const REDIS_KEY_PREFIX = 'enrichment:resolve:'
const REDIS_TTL = 600 // 10 minutes
// Exponential backoff: 2^n minutes, max 24h (in seconds)
const BACKOFF_BASE = 60 // 1 minute
const BACKOFF_MAX = 86400 // 24 hours

@Injectable()
export class EnrichmentService {
  private readonly logger = new Logger(EnrichmentService.name)

  constructor(
    private readonly providerRegistry: ProviderRegistry,
    private readonly repository: EnrichmentRepository,
    private readonly configsService: ConfigsService,
    private readonly redisService: RedisService,
  ) {}

  async resolve(url: string): Promise<{ result: EnrichmentResult; stale?: boolean }> {
    // 1. Check Redis cache
    const redisHit = await this.getFromRedis(url)
    if (redisHit) {
      return { result: redisHit }
    }

    // 2. Match URL to provider
    let parsedUrl: URL
    try {
      parsedUrl = new URL(url)
    } catch {
      throw new Error(`Invalid URL: ${url}`)
    }

    const matched = this.providerRegistry.match(parsedUrl)
    if (!matched) {
      throw new ProviderDisabledError('unknown')
    }

    const { provider, match } = matched

    // 3. Check provider enabled
    const config = await this.configsService.get('thirdPartyServiceIntegration')
    if (!this.isProviderEnabled(provider, config)) {
      throw new ProviderDisabledError(provider.name)
    }

    // 4. Check token
    if (provider.requiredConfigKeys?.length) {
      const hasToken = this.hasRequiredConfig(provider, config)
      if (!hasToken) {
        throw new TokenMissingError(provider.name)
      }
    }

    // 5. Check DB cache
    const dbRow = await this.repository.findByProviderAndExternalId(
      provider.name,
      match.id,
    )

    const now = new Date()

    if (dbRow) {
      const isExpired = dbRow.expiresAt && dbRow.expiresAt < now

      if (!isExpired) {
        // Fresh cache ŌĆö write to Redis and return
        await this.setToRedis(url, dbRow.normalized)
        return { result: dbRow.normalized }
      }

      // Expired ŌĆö check backoff
      const backoffSeconds = this.calculateBackoff(dbRow.failureCount)
      const backoffUntil = new Date(dbRow.fetchedAt.getTime() + backoffSeconds * 1000)

      if (dbRow.failureCount > 0 && now < backoffUntil) {
        // In backoff period ŌĆö return stale with warning
        return { result: dbRow.normalized, stale: true }
      }
    }

    // 6. Fetch fresh data
    try {
      const result = await provider.fetch(match.id)
      result.fetchedAt = new Date().toISOString()
      result.category = provider.category
      if (match.subtype) result.subtype = match.subtype

      const expiresAt = new Date(Date.now() + provider.defaultTtl * 1000)

      await this.repository.upsert(
        provider.name,
        match.id,
        match.fullUrl,
        result,
        null,
        expiresAt,
      )

      await this.setToRedis(url, result)
      return { result }
    } catch (error) {
      // 7. Fetch failed
      this.logger.warn(`Provider ${provider.name} fetch failed for ${match.id}: ${error.message}`)

      if (dbRow?.normalized) {
        // Record failure, return stale
        await this.repository.recordFailure(
          provider.name,
          match.id,
          error.message,
        )
        return { result: dbRow.normalized, stale: true }
      }

      throw error
    }
  }

  async getOne(providerName: string, id: string): Promise<EnrichmentResult> {
    const provider = this.providerRegistry.getByName(providerName)
    if (!provider) {
      throw new Error(`Unknown provider: ${providerName}`)
    }

    const row = await this.repository.findByProviderAndExternalId(providerName, id)
    if (row) {
      return row.normalized
    }

    // Not cached ŌĆö fetch fresh
    const result = await provider.fetch(id)
    result.fetchedAt = new Date().toISOString()
    result.category = provider.category

    const expiresAt = new Date(Date.now() + provider.defaultTtl * 1000)
    await this.repository.upsert(providerName, id, result.url, result, null, expiresAt)
    return result
  }

  async refresh(providerName: string, id: string): Promise<EnrichmentResult> {
    const provider = this.providerRegistry.getByName(providerName)
    if (!provider) {
      throw new Error(`Unknown provider: ${providerName}`)
    }

    const result = await provider.fetch(id)
    result.fetchedAt = new Date().toISOString()
    result.category = provider.category

    const expiresAt = new Date(Date.now() + provider.defaultTtl * 1000)
    await this.repository.upsert(providerName, id, result.url, result, null, expiresAt)

    // Invalidate Redis
    await this.deleteFromRedis(result.url)

    return result
  }

  async invalidate(providerName: string, id: string): Promise<void> {
    const row = await this.repository.findByProviderAndExternalId(providerName, id)
    if (row) {
      await this.deleteFromRedis(row.url)
      await this.repository.deleteByProviderAndExternalId(providerName, id)
    }
  }

  async list(page: number, size: number) {
    return this.repository.listPaginated(page, size)
  }

  getProviders(): ProviderMeta[] {
    return this.providerRegistry.getProviderMetas(() => true)
  }

  private async getFromRedis(url: string): Promise<EnrichmentResult | null> {
    const client = this.redisService.getClient()
    const key = this.redisKey(url)
    const cached = await client.get(key)
    if (!cached) return null
    try {
      return JSON.parse(cached) as EnrichmentResult
    } catch {
      return null
    }
  }

  private async setToRedis(url: string, result: EnrichmentResult): Promise<void> {
    const client = this.redisService.getClient()
    const key = this.redisKey(url)
    await client.set(key, JSON.stringify(result), 'EX', REDIS_TTL)
  }

  private async deleteFromRedis(url: string): Promise<void> {
    const client = this.redisService.getClient()
    const key = this.redisKey(url)
    await client.del(key)
  }

  private redisKey(url: string): string {
    const hash = createHash('sha1').update(url).digest('hex')
    return `${REDIS_KEY_PREFIX}${hash}`
  }

  private calculateBackoff(failureCount: number): number {
    return Math.min(BACKOFF_BASE * Math.pow(2, failureCount), BACKOFF_MAX)
  }

  private isProviderEnabled(provider: any, config: any): boolean {
    // Map provider names to config keys
    const configKeyMap: Record<string, string> = {
      'github': 'github',
      'tmdb': 'tmdb',
      'bangumi': 'bangumi',
      'neodb': 'neodb',
      'arxiv': 'arxiv',
      'leetcode': 'leetcode',
      'netease-music': 'neteaseMusic',
      'qq-music': 'qqMusic',
      'mx-space': 'self',
    }

    // Extract the base provider name from provider.name (e.g., 'gh-repo' ŌåÆ 'github')
    const category = provider.category
    const configKey = configKeyMap[category]

    if (!configKey || !config[configKey]) return true // No config ŌåÆ enabled by default

    return config[configKey].enabled !== false
  }

  private hasRequiredConfig(provider: any, config: any): boolean {
    const configKeyMap: Record<string, string> = {
      'github': 'github',
      'tmdb': 'tmdb',
      'bangumi': 'bangumi',
      'neodb': 'neodb',
      'arxiv': 'arxiv',
      'leetcode': 'leetcode',
      'netease-music': 'neteaseMusic',
      'qq-music': 'qqMusic',
    }

    const category = provider.category
    const configKey = configKeyMap[category]
    if (!configKey || !config[configKey]) return false

    const section = config[configKey]
    // Check if any token/key field has a non-empty value
    return Object.entries(section).some(
      ([key, val]) => key !== 'enabled' && typeof val === 'string' && val.length > 0,
    )
  }
}
```

```typescript
    }
    return res.json() as Promise<T>
  }
```

```typescript
  readonly name = 'arxiv'
  readonly displayName = 'Arxiv'
  readonly category = ENRICHMENT_CATEGORIES.ACADEMIC
  readonly priority = 10
  readonly defaultTtl = 86400 * 7 // 1 week ŌĆö papers don't change

  matchUrl(url: URL): UrlMatchResult | null {
    if (url.hostname !== 'arxiv.org') return null
    // arxiv.org/abs/2401.12345 or arxiv.org/pdf/2401.12345
    const match = url.pathname.match(/^\/(abs|pdf)\/([0-9.]+(?:v\d+)?)/)
    if (!match) return null
    return {
      id: match[2],
      fullUrl: `https://arxiv.org/abs/${match[2]}`,
      subtype: 'paper',
    }
  }

  isValidId(id: string): boolean {
    return /^[0-9.]+(?:v\d+)?$/.test(id)
  }

  async fetch(id: string): Promise<EnrichmentResult> {
    const res = await fetch(`https://export.arxiv.org/api/query?id_list=${id}`)
    if (!res.ok) throw new Error(`Arxiv API ${res.status}`)
    const text = await res.text()

    // Parse XML response
    const titleMatch = text.match(/<title>([\s\S]*?)<\/title>/g)
    const title = titleMatch?.[1]?.replace(/<[^>]+>/g, '').trim() || id
    const summaryMatch = text.match(/<summary>([\s\S]*?)<\/summary>/)
    const description = summaryMatch?.[1]?.replace(/\s+/g, ' ').trim()
    const authorMatch = text.match(/<author>[\s\S]*?<name>(.*?)<\/name>/g)
    const authors = authorMatch?.map((a) => {
      const m = a.match(/<name>(.*?)<\/name>/)
      return m?.[1] || ''
    })

    const attrs: EnrichmentResult['attributes'] = []
    if (authors?.length)
      attrs.push({
        key: 'authors',
        value: authors.slice(0, 5).join(', ') + (authors.length > 5 ? ' et al.' : ''),
        label: 'Authors',
        format: 'text',

      title,
      description,
      url: `https://arxiv.org/abs/${id}`,
```

```typescript
  readonly displayName = 'Mix Space'

  readonly priority = 5
  readonly defaultTtl = 300 // 5 minutes ŌĆö internal content changes

  constructor(private readonly databaseService: DatabaseService) {}

  matchUrl(url: URL): UrlMatchResult | null {
    // Match against configured server URL
    // e.g., serverUrl/posts/slug-here or serverUrl/notes/nid
    // This provider is special ŌĆö it resolves internal content URLs
    const path = url.pathname
    const postMatch = path.match(/^\/posts\/([^/]+)/)
    if (postMatch) return { id: `post:${postMatch[1]}`, fullUrl: url.href, subtype: 'post' }

    const noteMatch = path.match(/^\/notes\/(\d+)/)
    if (noteMatch) return { id: `note:${noteMatch[1]}`, fullUrl: url.href, subtype: 'note' }

    return null
  }

  isValidId(id: string): boolean {
    return /^(post|note):/.test(id)
  }

  async fetch(id: string): Promise<EnrichmentResult> {
    const [type, ...rest] = id.split(':')
    const slugOrNid = rest.join(':')

    if (type === 'post') {
      const doc = await this.databaseService.findGlobalById(slugOrNid)
      if (!doc) throw new Error(`Post not found: ${slugOrNid}`)
      const item = doc.posts?.[0]
      if (!item) throw new Error(`Post not found: ${slugOrNid}`)

      return {
        title: item.title || slugOrNid,
        description: (item.text || '').slice(0, 300) || undefined,
        url: id,
        category: this.category,
        subtype: 'post',
        publishedAt: item.created ? new Date(item.created).toISOString() : undefined,
        fetchedAt: '',
        attributes: [
          { key: 'type', value: 'post', label: 'Type', format: 'text' },
        ],
      }
    }

    if (type === 'note') {
      const nid = parseInt(slugOrNid, 10)
      const doc = await this.databaseService.findGlobalById(String(nid))
      if (!doc) throw new Error(`Note not found: ${nid}`)
      const item = doc.notes?.[0]
      if (!item) throw new Error(`Note not found: ${nid}`)

        attributes: [
          { key: 'type', value: 'note', label: 'Type', format: 'text' },
        ],
      }
    }
```

```typescript
private async attachEnrichment<T extends RecentlyRow>(
  rows: T[],
): Promise<Array<T & { enrichment?: EnrichmentResult | null }>> {
  const rowsWithRef = rows.filter(
    (r) => r.enrichmentProvider && r.enrichmentExternalId,
  )
  if (rowsWithRef.length === 0) {
    // Fallback for old rows without enrichment reference
    return rows.map((row) => ({
      ...row,
      enrichment: row.metadata
        ? this.convertLegacyMetadata(row.metadata, row.type)
        : null,
    }))
  }

  // Batch load from enrichment_cache
  const enrichmentMap = new Map<string, EnrichmentResult>()
  for (const row of rowsWithRef) {
    try {
      const result = await this.enrichmentService.getOne(
        row.enrichmentProvider!,
        row.enrichmentExternalId!,
      )
      enrichmentMap.set(`${row.enrichmentProvider}:${row.enrichmentExternalId}`, result)
    } catch {
      // Skip failed enrichments
    }
  }

  return rows.map((row) => ({
    ...row,
    enrichment:
      row.enrichmentProvider && row.enrichmentExternalId
        ? enrichmentMap.get(`${row.enrichmentProvider}:${row.enrichmentExternalId}`) || null
        : row.metadata
          ? this.convertLegacyMetadata(row.metadata, row.type)
          : null,
  }))
}
```

> **Note:** `convertLegacyMetadata` is a helper that converts the old `metadata` jsonb into an `EnrichmentResult` shape. This provides backward compatibility during the migration period.
